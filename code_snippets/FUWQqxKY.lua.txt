function Autofarm(W)
    spawn(function ()
        while getgenv().Autofarm do
            for i, v in pairs(game:GetService("Workspace").Monsters[W]:GetChildren()) do
                TPCFrame(v.RootPart.CFrame) wait(0.2)
                local args = {[1] = "OneKillMonster",[2] = v}game:GetService("ReplicatedStorage").RemoteEvents.MonsterRemote:FireServer(unpack(args)) wait()
            end wait()
        end
    end)
    spawn(function ()
        while getgenv().Autofarm do
local args = {[1] = "UnequipAllFollow"} game:GetService("ReplicatedStorage").RemoteEvents.PlayerDataRemote:FireServer(unpack(args)) wait()
local args = {[1] = "BestEquipRole"} game:GetService("ReplicatedStorage").RemoteEvents.PlayerDataRemote:FireServer(unpack(args)) wait()
        end
    end)
end

end
 
function Speed(Nume)
    spawn(function ()
        if getgenv().Speed == true then
            while getgenv().Speed == true do
                game:GetService("Players").LocalPlayer.Character.Humanoid.WalkSpeed = Nume
                wait()
            end
        elseif getgenv().Speed == false then
            game:GetService("Players").LocalPlayer.Character.Humanoid.WalkSpeed = 16
        end
    end)
end
