CreateConVar( "sv_player_sounds", "1", { FCVAR_ARCHIVE, FCVAR_REPLICATED, FCVAR_NOTIFY } ) 
 
--sonidos de muerte
 
local function playerDeathSounds(ply)
    if GetConVarNumber( "sv_player_sounds" ) == 1 then
    
    if (ply:GetModel() == "models/player.mdl") then
        ply:EmitSound("player/pain" .. math.random( 1, 14 ) .. ".mp3") end
    end
end
hook.Add( "DoPlayerDeath", "playerDeathSounds", playerDeathSounds )
 
--sonidos de daño
 
local function playerHurtSounds(ply)
    if GetConVarNumber( "sv_player_sounds" ) == 1 and math.random(1,2)==2 then
    if ply:Health() == 0 then return end
 
    if (ply:GetModel() == "models/player.mdl") then
    ply:EmitSound("player/pain" .. math.random( 1, 14 ) .. ".mp3")
    end
 
    end
end
 
--sonido al aparecer
hook.Add( "PlayerHurt", "playerHurtSounds", playerHurtSounds )
 
if (ply:GetModel() == "models/player.mdl") then
ply:EmitSound("player/playerspawn" .. math.random( 1, 11 ) .. ".mp3")
end
 
end
end)
 
--sonidos al matar o ser asesinado
 
sound.AddSoundOverrides( "lua/player_name/player_murder_sounds.lua" )
 
hook.Add("SetupMove", "player_first", function(ply)
    if(ply:GetModel() == "models/player.mdl") then
        ply.player_first = true
        else
        ply.player_first = false
    end
end)
 
hook.Add("OnNPCKilled", "player_first_NPCKillTaunt", function(victim, killer, weapon)
    if GetConVarNumber( "sv_player_sounds" ) == 1 then
    if (( killer:IsPlayer() and killer.player_first == true and math.random(1,3)==3 )) then
        killer:EmitSound("player_first_player.kill")
        timer.Simple(math.random(0.8,1), function()
        end)
    end
    end
end)

hook.Add("KeyPress", "player_first_Jump", function(ply, key)
    if GetConVarNumber( "sv_player_sounds" ) == 1 then
    if ( SERVER and ply:Alive() and key==IN_JUMP and ply:IsOnGround() and !ply:InVehicle() and ply:GetMoveType() != MOVETYPE_NOCLIP) then
        if CurTime()<(ply.spyro_Spawn_Last or -999)+0.2 then return end
        if ply.player_first then
            local sndtbl = ply.player_first
            if sndtbl then
                ply:EmitSound( "nombre_del_salto" .. math.random( 1, 3 ) .. ".mp3" )
                return true
            end
